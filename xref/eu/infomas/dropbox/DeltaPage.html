23  package eu.infomas.dropbox;
24  
25  import java.io.Serializable;
26  import java.util.ArrayList;
27  import java.util.Collections;
28  import java.util.List;
29  import java.util.Map;
30  
31  import static eu.infomas.dropbox.Utils.*;
32  
33  /**
34   * A page of {@link DeltaEntry}s, returned by {@link Dropbox#delta(java.lang.String)}.
35   *
36   * @author Original Author is Dropbox
37   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a> (refactoring)
38   * @since infomas-asl 3.0.2
39   */
40  public final class DeltaPage implements Serializable {
41  
42      private static final long serialVersionUID = 1L;
43  
44      private final String cursor;
45      private final boolean reset;
46      private final List<DeltaEntry> entries;
47      private final boolean hasMore;
48  
49      DeltaPage(final Map<String, Object> jsonMap) {
50          reset = asBoolean(jsonMap, "reset");
51          cursor = asString(jsonMap, "cursor");
52          hasMore = asBoolean(jsonMap, "hasMore");
53          final List<List<Object>> list = (List<List<Object>>) jsonMap.get("entries");
54          entries = new ArrayList<DeltaEntry>(list.size());
55          for (final List<Object> entryList : list) {
56              entries.add(new DeltaEntry(entryList));
57          }
58      }
59  
60      /**
61       * A string that is used to keep track of your current state. 
62       * On the next call to {@link Dropbox#delta(java.lang.String) }, 
63       * pass in this value to pick up where you left off.
64       */
65      public String getCursor() {
66          return cursor;
67      }
68  
69      /**
70       * If <code>true</code>, then you should reset your local state to be an empty folder
71       * before processing the list of delta entries. This is only <code>true</code> in 
72       * rare situations.
73       */
74      public boolean isReset() {
75          return reset;
76      }
77  
78      /**
79       * Apply these entries to your local state to catch up with the Dropbox server's
80       * state.
81       */
82      public List<DeltaEntry> getEntries() {
83          return Collections.unmodifiableList(entries);
84      }
85  
86      /**
87       * If <code>true</code>, then there are more entries available; you can call {@link
88       * Dropbox#delta(java.lang.String) } again immediately to retrieve those entries. If
89       * <code>false</code>, then wait at least 5 minutes (preferably longer) before
90       * checking again.
91       */
92      public boolean hasMore() {
93          return hasMore;
94      }
95      
96  }
