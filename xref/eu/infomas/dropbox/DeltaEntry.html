23  package eu.infomas.dropbox;

25  import java.io.Serializable;
26  import java.util.List;
27  import java.util.Map;

29  /**
30   * A single entry in a {@link DeltaPage}.
31   *
32   * @author Original Author is Dropbox
33   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a> (refactoring)
34   * @since infomas-asl 3.0.2
35   */
36  public final class DeltaEntry implements Serializable {
37  
38      private static final long serialVersionUID = 1L;
39  
40      private final String lcPath;
41      private final Entry metadata;
42  
43      DeltaEntry(final List<Object> jsonList) {
44          this.lcPath = (String) jsonList.get(0);
45          this.metadata = Entry.valueOf((Map) jsonList.get(1));
46      }
47  
48      /**
49       * The lower-cased path of the entry. 
50       * Dropbox compares file paths in a case-insensitive manner. For example, 
51       * an entry for <code>"/readme.txt"</code> should overwrite the entry for
52       * <code>"/ReadMe.TXT"</code>.
53       * <p>
54       * To get the original case-preserved path, look in the {@link #metadata metadata}
55       * field.
56       */
57      public String getLowerCasedPath() {
58          return lcPath;
59      }
60  
61      /**
62       * If this is
63       * <code>null</code>, it means that this path doesn't exist on on Dropbox's copy of
64       * the file system. To update your local state to match, delete whatever is at that
65       * path, including any children. If your local state doesn't have anything at this
66       * path, ignore this entry.
67       * <p>
68       * If this is not <code>null</code>, it means that Dropbox has a file/folder at this 
69       * path with the given metadata. To update your local state to match, add the entry 
70       * to your local state as well.
71       * <ul> 
72       * <li> If the path refers to parent folders that don't exist
73       * yet in your local state, create those parent folders in your local state.</li>
74       * <li> If the metadata is for a file, replace whatever your local state has at that
75       * path with the new entry.</li>
76       * <li> If the metadata is for a folder, check what your local state has at the path. 
77       * If it's a file, replace it with the new entry. If it's a folder, apply the new 
78       * metadata to the folder, but do not modify the folder's
79       * children.</li>
80       * </ul>
81       */
82      public Entry getMetadata() {
83          return metadata;
84      }
85      
86  }
