22  package eu.infomas.dropbox;
23  
24  import java.io.Serializable;
25  
26  /**
27   * {@code Credentials} holds the key (identifier) and secret token pair used in the OAuth
28   * protocol and REST requests.
29   *
30   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
31   * @since infomas-asl 3.0.2
32   */
33  public final class Credentials implements Serializable {
34      
35      private static final long serialVersionUID = 1L;
36  
37      private final String key;
38      private final String secret;
39  
40      /**
41       * See factory method {@link Credentials#of(java.lang.String, java.lang.String) }.
42       */
43      private Credentials(final String key, final String secret) {
44          this.key = Utils.notNullOrBlank("key", key);
45          this.secret = Utils.notNullOrBlank("secret", secret);
46      }
47  
48      /**
49       * Factory method. Create a new {@code Credentials} object. 
50       * If both 'key' and 'secret' are {@code null}, {@code null} is returned. 
51       * Otherwise both 'key' and 'secret' must be non-{@code null} and non-empty Strings.
52       *
53       * @throws IllegalArgumentException if key or secret is {@code null} or empty.
54       */
55      public static Credentials of(final String key, final String secret) {
56          if (key == null && secret == null) {
57              return null;
58          }
59          return new Credentials(key, secret);
60      }
61  
62      /**
63       * Return the key (identifier) of this {@code Credentials}.
64       */
65      public String getKey() {
66          return key;
67      }
68  
69      /**
70       * Return the secret token of this {@code Credentials}.
71       */
72      public String getSecret() {
73          return secret;
74      }
75  
76      @Override
77      public int hashCode() {
78          return key.hashCode() ^ secret.hashCode();
79      }
80  
81      @Override
82      public boolean equals(final Object other) {
83          if (other == this) {
84              return true;
85          }
86          if (other instanceof Credentials) {
87              Credentials t = (Credentials) other;
88              return t.key.equals(key) && t.secret.equals(secret);
89          }
90          return false;
91      }
92  
93      @Override
94      public String toString() {
95          return "Credentials[key=" + key + ", secret=" + secret + ']';
96      }
97      
98  }
