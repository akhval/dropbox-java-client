41  /**
42   * {@code DropBox} offers an easy-to-use "fluent" interface for working with the 
43   * <a href="https://www.dropbox.com/developers/reference/api">Dropbox REST API v1</a>.
44   * All operations (REST Requests) are done via this facade API.
45   * <p>
46   * Dropbox does use <a href="http://tools.ietf.org/html/rfc5849">the OAuth 1.0
47   * Protocol</a> which is also supported by this class.
48   * <p>
49   * The actual HTTP communication is delegated to a {@link RestClient} implementation, which 
50   * can easily use a different implementation than the default provided by using the
51   * Java {@link java.util.ServiceLoader service-provider loading facility}.
52   * <p>
53   * <b>NOTE</b> that this class is <b>not</b> thread-safe!
54   * {@code Dropbox} instances are cheap to create, so do not cache these instances,
55   * just create a new instance when needed, use it and throw it away!
56   * 
57   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
58   * @since infomas-asl 3.0.2
59   */

119     /**
120      * Create a new {@code Dropbox} instance.
121      * This is a low level interface, in practice you mostly will use
122      * {@link #Dropbox(String)} or {@link #Dropbox(Properties)}.
123      * 
124      * @param clientCredentials The client (application) credentials. Mandatory.
125      * @param tokenCredentials The token (access or request) credentials, may be 
126      * {@code null} in which case the token credentials must be requested by using
127      * {@link #requestTemporaryCredentials() }
128      */
129     public Dropbox(final Credentials clientCredentials, final Credentials tokenCredentials) {
130         this.clientCredentials = clientCredentials;
131         if (tokenCredentials != null) {
132             setClientCredentials(tokenCredentials);
133         }
134     }
135     
136     /**
137      * Create a new {@code Dropbox} instance, using the configuration data supplied
138      * by the {@code Properties} object.
139      * The following property names are recognized:
140      * <ul>
141      * <li>{@link #CLIENT_CREDENTIALS_KEY}</li>
142      * <li>{@link #CLIENT_CREDENTIALS_SECRET}</li>
143      * <li>{@link #TOKEN_CREDENTIALS_KEY}</li>
144      * <li>{@link #TOKEN_CREDENTIALS_SECRET}</li>
145      * <li>{@link #LANGUAGE_KEY}</li>
146      * </ul>
147      */
148     public Dropbox(final Properties configuration) {
149         this(
150             Credentials.of(
151                 configuration.getProperty(CLIENT_CREDENTIALS_KEY),
152                 configuration.getProperty(CLIENT_CREDENTIALS_SECRET)),
153             Credentials.of(
154                 configuration.getProperty(TOKEN_CREDENTIALS_KEY),
155                 configuration.getProperty(TOKEN_CREDENTIALS_SECRET))
156         );
157         if (configuration.containsKey(LANGUAGE_KEY)) {
158             this.locale = new Locale(configuration.getProperty(LANGUAGE_KEY));
159         }
160     }

162     /**
163      * Create a new {@code Dropbox} instance, using the configuration data supplied
164      * by the {@code Properties} file found at the specified location.
165      * 
166      * @param configurationPath The path where the configuration {@code Properties} file
167      * can be found.
168      * If a relative 'configurationPath' is specified, the configuration is loaded 
169      * relative to the current user ({@code user.dir}) directory. If the path starts 
170      * with "~/" the configuration file is loaded relative to the current user home 
171      * ({@code user.home}) directory.
172      * If the prefix {@code classpath:} is used, the 'configurationPath' is loaded from 
173      * the class path. 
174      * May not be empty or {@code null}.
175      * 
176      * @see #Dropbox(java.util.Properties) 
177      */
178     public Dropbox(final String configurationPath) throws IOException {
179         this(loadProperties(configurationPath, Dropbox.class));
180     }

198     /**
199      * After a {@code Dropbox} client is created, you are able to set the request or
200      * token {@code Credentials} once. These {@code Credentials} are returned by 
201      * {@link #requestTokenCredentials(eu.infomas.dropbox.Credentials) }.
202      */
203     public void setClientCredentials(final Credentials tokenCredentials) {
204         if (tokenCredentials == null) {
205             throw new IllegalArgumentException("'tokenCredentials' is null");
206         }
207         if (this.authorization != null) {
208             throw new IllegalStateException("Token Credentials already set.");
209         }
210         authorization = getAuthorization(clientCredentials, tokenCredentials);
211     }

213     // Authentication ===================================================================
214     
215     /**
216      * First step in the OAuth 1.0 protocol: Temporary Credentials Request. In this
217      * request the client credentials are used which must be provided by Dropbox to this
218      * application in advance.
219      *
220      * @return Temporary Credentials
221      */
222     public Credentials requestTemporaryCredentials() throws IOException {
223         String response = request("GET", API_SERVER, "/oauth/request_token")
224             .withHeader("Authorization", String.format(
225                 "OAuth oauth_version=\"1.0\", " +
226                 "oauth_signature_method=\"PLAINTEXT\", " +
227                 "oauth_consumer_key=\"%s\", " +
228                 "oauth_signature=\"%s&\"", 
229                 clientCredentials.getKey(), 
230                 clientCredentials.getSecret()))
231             .asString(restClient);
232         return parseCredentials(response);
233     }
234 
235     /**
236      * Second step in the OAuth 1.0 protocol: Resource Owner Authorization Request.
237      * Redirect the Resource Owner (User) to the endpoint returned by this method. If the
238      * Resource Owner approved the access to the resource(s), the Token Credentials can be
239      * requested by
240      * {@link #requestTokenCredentials(eu.infomas.dropbox.Credentials) }.
241      *
242      * @param temporaryCredentials The Temporary Credentials as returned by
243      * {@link #requestTemporaryCredentials() }
244      *
245      * @return The endpoint URI for the Resource Owner Authorization Request
246      */
247     public URL getResourceOwnerAuthorizationEndpoint(final Credentials temporaryCredentials)
248         throws IOException {
249 
250         return request("GET", API_SERVER, "/oauth/authorize")
251             .withParameter("oauth_token", temporaryCredentials.getKey())
252             .toURL();
253     }
254 
255     /**
256      * Third step in the OAuth 1.0 protocol: Token Credentials Request.
257      *
258      * @param temporaryCredentials The Temporary Credentials as returned by
259      * {@link #requestTemporaryCredentials() }
260      *
261      * @return The Token Credentials which can be used to access the restricted resources
262      * of the Resource Owner
263      */
264     public Credentials requestTokenCredentials(final Credentials temporaryCredentials)
265         throws IOException {
266 
267         String response = request("GET", API_SERVER, "/oauth/access_token")
268             .withHeader("Authorization", 
269                 getAuthorization(clientCredentials, temporaryCredentials))
270             .asString(restClient);
271         // we ignore the user id (uid) because we do not need it and it can be requested
272         // by accountInfo()
273         return parseCredentials(response);
274     }

297     /**
298      * Builder for the {@link #filesGet(java.lang.String)} service.
299      */
300     public static final class FilesGet extends AbstractBuilder {
301         private String rev;
302         private int rangeFirst;
303         private int rangeLast;
304         
305         private FilesGet(final Dropbox dropbox, final String path) {
306             super(dropbox, path);
307         }
308         
309         /**
310          * Specify the revision of the file to retrieve ({@code rev}). 
311          * If not specified, the most recent revision is used.
312          * 
313          * @return The {@code FilesGet} Builder
314          */
315         public FilesGet withRev(final String rev) {
316             this.rev = rev;
317             return this;
318         }
319         
320         /**
321          * Specify the byte range to retrieve.
322          * If not specified, the complete file is retrieved.
323          * 
324          * @param first Index position of first byte to retrieve (zero based, inclusive)
325          * @param last Index position of last byte to retrieve (zero based, inclusive)
326          * 
327          * @return The {@code FilesGet} Builder
328          */
329         public FilesGet withRange(final int first, final int last) {
330             if (first < 0) {
331                 throw new IllegalArgumentException("'first' < 0: " + first);
332             }
333             if (first >= last) {
334                 throw new IllegalArgumentException(
335                     "'first' >= 'last': " + first + " >= " + last);
336             }
337             this.rangeFirst = first;
338             this.rangeLast = last;
339             return this;
340         }
341         
342         /**
343          * Call the service and write the requested file to the specified {@code File}.
344          * 
345          * @return The bytes written to the file (file size)
346          */
347         public long toFile(final File file) throws IOException {
348             return toOutputStream(new FileOutputStream(file));
349         }
350         
351         /**
352          * Call the service and write the requested file to the specified output
353          * stream.
354          * 
355          * @return The bytes written to the output stream
356          */        
357         public long toOutputStream(final OutputStream out) throws IOException {
358             Request.Builder builder = dropbox
359                 .request("GET", CONTENT_SERVER, "/files/sandbox" + path)
360                 .withHeader("Authorization", dropbox.authorization)
361                 .withParameter("rev", rev);
362             if (rangeLast > 0) {
363                 builder.withHeader("Range", "bytes=" + rangeFirst + '-' + rangeLast);
364             }
365             return builder.toOutputStream(dropbox.restClient, out);
366         }
367     }

393     public static class FilesPut extends AbstractBuilder {
394 
395         private boolean overwrite;
396         private String parentRev = "";
397 
398         FilesPut(final Dropbox dropbox, final String path) {
399             super(dropbox, path);
400         }
401                 
402         /**
403          * Specify the revision of the file you're editing (parent_rev).
404          * The revision may be empty, but not {@code null}.
405          */
406         public FilesPut withParentRev(final String parentRev) {
407             Utils.notNull("parentRev", parentRev);
408             this.parentRev = parentRev;
409             return this;
410         }
411 
412         /**
413          * Used by subclasses.
414          */
415         String getParentRev() {
416             return parentRev;
417         }
418         
419         /**
420          * Specify that an already existing file can be overwritten (overwrite).
421          * Default is that the file is not overwritten, but a new file name is created.
422          */
423         public FilesPut withOverwrite() {
424             this.overwrite = true;
425             return this;
426         }
427         
428         /**
429          * Used by subclasses.
430          */
431         boolean isOverwrite() {
432             return overwrite;
433         }
434         
435         /**
436          * Call the service and upload the data from the specified input stream.
437          * Note that a length must be specified!
438          * Maximum file size limit is 150 MB.
439          */
440         public Entry fromInputStream(final InputStream is, final long length) 
441             throws IOException {
442             
443             final String response = dropbox
444                 .request("PUT", CONTENT_SERVER, 
445                 "/files_put/sandbox" + (path.startsWith("/") ? path : "/" + path))
446                 .withHeader("Authorization", dropbox.authorization)
447                 .withHeader("Content-Length", Long.toString(length))
448                 .withParameter("overwrite", overwrite)
449                 .withParameter("parent_rev", parentRev) // maybe empty, not null
450                 .withPayload(is)
451                 .asString(dropbox.restClient);
452             return Entry.valueOf(parseJson(response, Map.class));
453         }
454         
455         /**
456          * Call the service and upload the data from the specified file.
457          * Maximum file size limit is 150 MB.
458          */
459         public Entry fromFile(final File file) throws IOException {
460             return fromInputStream(new FileInputStream(file), file.length());
461         }

483      * Builder for the {@link #metadata(java.lang.String)} service.
484      */
485     public static final class Metadata extends AbstractBuilder {
486         private int fileLimit = 25000; // default value
487         private String hash;
488         private boolean list;
489         private String rev;
490 
491         private Metadata(final Dropbox dropbox, final String path) {
492             super(dropbox, path);
493         }
494         
495         /**
496          * Specify the revision to use ({@code rev}). 
497          * Optional.
498          */
499         public Metadata withRev(final String rev) {
500             this.rev = rev;
501             return this;
502         }
503         
504         /**
505          * Specify the file limit to use ({@code file_limit}). 
506          * Optional.
507          * Default value is 25000.
508          */
509         public Metadata withFileLimit(final int fileLimit) {
510             this.fileLimit = fileLimit;
511             return this;
512         }
513         
514         /**
515          * Specify the hash to use ({@code hash}). 
516          * Optional.
517          */
518         public Metadata withHash(final String hash) {
519             this.hash = hash;
520             return this;
521         }
522         
523         /**
524          * Specify the children (files directories) must be returned when the queried
525          * path is a directory ({@code list}).
526          * Optional, default {@code false}.
527          */
528         public Metadata withList() {
529             this.list = true;
530             return this;
531         }
532         
533         /**
534          * Call the service and return the response as a JSON String.
535          * 
536          * @see #asEntry() 
537          */        
538         public String asJson() throws IOException {
539             final String response = dropbox
540                 .request("GET", API_SERVER, "/metadata/sandbox" + path)
541                 .withHeader("Authorization", dropbox.authorization)
542                 .withParameter("file_limit", fileLimit)
543                 .withParameter("hash", hash)
544                 .withParameter("list", list)
545                 .withParameter("rev", rev)
546                 .withParameter("file_limit", fileLimit)
547                 .asString(dropbox.restClient);
548             if (response == null) {
549                 LOG.log(Level.WARNING, 
550                     "Response is null for path ''/metadata/sandbox/{0}''", path);
551             }
552             return response;
553         }
554         
555         /**
556          * Call the service and return the response as an {@link Entry} object.
557          * 
558          * @see #asJson() 
559          */
560         public Entry asEntry() throws IOException {
561             final String json = asJson();
562             final Map<String, Object> map = parseJson(json, Map.class);
563             if (map == null) {
564                 LOG.log(Level.WARNING, "Map is null for path ''{0}'', content=''{1}''", 
565                     new Object[] {path, json});
566             }
567             return Entry.valueOf(map);
568         }
569     }

608     /**
609      * Obtains metadata for the previous revisions of a file.
610      * <p>
611      * This method wraps the 
612      * <a href="https://www.dropbox.com/developers/reference/api#revisions">/revisions</a> 
613      * service. See the official API Documentation for more information.
614      * 
615      * @param path On the first call, you should pass in {@code null}.  
616      *               On subsequent calls, pass in the {@link DeltaPage#cursor cursor} 
617      *               returned by the previous call
618      * @param limit rev_limit Default is 10. Max is 1,000. When listing a file, the 
619      * service will not report listings containing more than the amount specified and 
620      * will instead respond with a 406 (Not Acceptable) status response.
621      */
622     public List<Entry> revisions(final String path, final int limit) 
623         throws IOException {
624         
625         final String response = request("GET", API_SERVER, "/revisions/sandbox/" + path)
626             .withHeader("Authorization", authorization)
627             .withParameter("rev_limit", limit)
628             .asString(restClient);
629         
630         final List<Map<String, Object>> jsonList = 
631             (List<Map<String, Object>>)parseJson(response, List.class);
632         final List<Entry> entries = new ArrayList<Entry>(jsonList.size());
633         for (final Map<String, Object> jsonMap : jsonList) {
634             final Entry entry = Entry.valueOf(jsonMap);
635             if (entry != null) {
636                 entries.add(entry);
637             }
638         }
639         //writeContentToFile(response, new File("./revisions.json"));
640         return entries;
641     }

695      */    
696     // http://stackoverflow.com/questions/5346726/java-inheritance-using-builder-pattern
697     public static final class ChunkedFilesPut extends FilesPut {
698         
699         private int chunkSize;
700         
701         private ChunkedFilesPut(final Dropbox dropbox, final String path) {
702             super(dropbox, path);
703             withChunkSize(4);
704         }
705         
706         /**
707          * Specify the chuck size in MB.
708          * If not specified, 4 MB is used as default.
709          * The maximum chunk size is 150 MB.
710          */
711         public ChunkedFilesPut withChunkSize(final int chunkSize) {
712             if (chunkSize <= 0 || chunkSize > 150) {
713                 throw new IllegalArgumentException("Invalid chunk size: " + chunkSize);
714             }
715             this.chunkSize = chunkSize * 1024 * 1024;
716             return this;
717         }
718         
719         /**
720          * Call the service and upload the data from the specified input stream.
721          * Note that a length must be specified!
722          * Maximum file size limit is 150 MB.
723          */
724         @Override
725         @SuppressWarnings("unchecked")
726         public Entry fromInputStream(final InputStream is, final long length) 
727             throws IOException {
728 
729             // https://www.dropbox.com/developers/core/docs#chunked-upload
730             final ChunkedInputStream chunked = new ChunkedInputStream(is, chunkSize);
731             String uploadId = null; // at first PUT, uploadId should not be included
732             long offset = 0;
733             while (chunked.nextChunk()) {
734                 // response holds JSON String with "upload_id", "offset" and "expires"
735                 final String response = dropbox
736                     .request("PUT", CONTENT_SERVER, "/chunked_upload")
737                     .withHeader("Authorization", dropbox.authorization)
738                     .withParameter("upload_id", uploadId)
739                     .withParameter("offset", offset)
740                     .withPayload(chunked)
741                     .asString(dropbox.restClient);
742                 final Map<String, Object> json = parseJson(response, Map.class);
743                 uploadId = asString(json, "upload_id");
744                 offset = asLong(json, "offset");
745             }
746             // ChunkedInputStream handles close() itself
747             // https://www.dropbox.com/developers/core/docs#commit-chunked-upload
748             final String response = dropbox.request("POST", CONTENT_SERVER, 
749                 "/commit_chunked_upload/sandbox" + (path.startsWith("/") ? path : "/" + path))
750                 .withHeader("Authorization", dropbox.authorization)
751                 .withParameter("upload_id", uploadId)
752                 .withParameter("parent_rev", getParentRev())
753                 .withParameter("overwrite", isOverwrite())
754                 .asString(dropbox.restClient);
755             return Entry.valueOf(parseJson(response, Map.class));
756         }
757         
758     }
