34  /**
35   * {@code Entry} describes the meta data of a Dropbox file or folder. It is just a simple
36   * POJO, offering a type safe interface to the JSON response returned by the Dropbox REST
37   * API.

63      private Entry(final Map<String, Object> jsonMap) {
64          bytes = asLong(jsonMap, "bytes");
65          hash = asString(jsonMap, "hash");
66          icon = asString(jsonMap, "icon");
67          isDir = asBoolean(jsonMap, "is_dir");
68          modified = asDate(jsonMap, "modified");
69          clientMtime = asString(jsonMap, "client_mtime");
70          path = asString(jsonMap, "path");
71          root = asString(jsonMap, "root");
72          size = asString(jsonMap, "size");
73          mimeType = asString(jsonMap, "mime_type");
74          rev = asString(jsonMap, "rev");
75          revision = asLong(jsonMap, "revision");
76          thumbExists = asBoolean(jsonMap, "thumb_exists");
77          isDeleted = asBoolean(jsonMap, "is_deleted");
78  
79          final List<Map<String, Object>> array = 
80              (List<Map<String, Object>>)jsonMap.get("contents");
81          if (array != null) {
82              final List<Entry> tmp = new ArrayList<Entry>(array.size());
83              for (final Map<String, Object> element : array) {
84                  tmp.add(new Entry(element));
85              }
86              contents = Collections.unmodifiableList(tmp);
87          } else {
88              contents = Collections.emptyList();
89          }
90      }

94       * provided as {@link Map}.

134     static Entry valueOf(final Map<String, Object> jsonMap) {
135         return jsonMap == null ? null : new Entry(jsonMap);

265     /**
266      * Returns the path of the parent directory if this is a file.
267      */
268     public String parentPath() {
269         if (path.equals("/")) {
270             return "";
271         } else {
272             final int index = path.lastIndexOf('/');
273             return path.substring(0, index + 1);
274         }
275     }
