22  package eu.infomas.dropbox;
23  
24  import java.io.IOException;
25  import java.io.OutputStream;
26  import java.util.Iterator;
27  import java.util.ServiceLoader;
28  import java.util.logging.Level;
29  import java.util.logging.Logger;
30  
31  /**
32   * {@code RestClient} defines the interface for a simple HTTP(S) REST client, used by
33   * {@link Dropbox} and {@link Request}. 
34   * It is <b>not</b> a general purpose REST Client!
35   * <p>
36   * Functionality is very simple and "just enough" for getting {@link Dropbox} working. 
37   * Actual implementations must extend this class and are discovered via the standard Java
38   * {@link ServiceLoader service-provider loading} facility.
39   * <p>
40   * Note that this type is part of the low-level API. In most uses cases you do not need 
41   * this class (interfacing is done via {@link Dropbox} API).
42   *
43   * @author <a href="mailto:rmuller@xiam.nl">Ronald K. Muller</a>
44   * @since infomas-asl 3.0.2
45   */
46  public abstract class RestClient {
47  
48      private static final Logger LOG = Logger.getLogger(RestClient.class.getName());
49  
50      /**
51       * Return an concrete implementation of a {@code RestClient}. The standard Java
52       * {@link ServiceLoader service-provider loading facility} is used. If no
53       * implementation is found, {@link SimpleRestClient} is used. If several
54       * implementations are available, the first found on the class path is used and a
55       * warning is logged. 
56       * <p>
57       * Note that this factory method always returns a newly created instance.
58       */
59      public static RestClient newInstance() {
60          final Iterator<RestClient> clients = ServiceLoader.load(RestClient.class).iterator();
61          final RestClient client;
62          if (clients.hasNext()) {
63              client  = clients.next();
64              while (clients.hasNext()) {
65                  LOG.log(Level.WARNING, "RestClient implementation SKIPPED: {0}",
66                      clients.next().getClass().getName());
67              }
68              return client;
69          } else {
70              client = new SimpleRestClient();
71          }
72          LOG.log(Level.INFO, "RestClient implementation used: {0}.", 
73              client.getClass().getName());
74          return client;
75      }
76      
77      /**
78       * Execute the specified {@link Request} and return the response body (payload) as
79       * a String.
80       * 
81       * @param request The request
82       * 
83       * @return The response message body as String
84       */
85      public abstract String asString(final Request request) 
86          throws IOException;
87      
88      /**
89       * Execute the specified {@link Request} and write the response body (payload) to the 
90       * specified OurputStream.
91       * 
92       * @param request The request
93       * @param responseStream The data from the response message body (payload)
94       * 
95       * @return The bytes written to the output stream
96       */    
97      public abstract long toOutputStream(final Request request, 
98          final OutputStream responseStream) throws IOException;
99
